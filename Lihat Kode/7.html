      <button onclick="konversiMataUang()">Konversi</button>

    <script>
      function konversiMataUang() {
        const nominal = parseFloat(document.getElementById("nominal").value);
        const jenis = document.getElementById("jenis").value;
        const output = document.getElementById("output");
        let hasil = "";

        if (isNaN(nominal) || nominal < 0) {
          hasil = "Masukkan nominal yang valid.";
        } else {
          const rateUSD_IDR = 15000;
          const rateYEN_IDR = 120;

          switch (jenis) {
            case "USDtoIDR":
              hasil = nominal + " USD = " + (nominal * rateUSD_IDR).toLocaleString("id-ID") + " IDR";
              break;
            case "YENtoIDR":
              hasil = nominal + " YEN = " + (nominal * rateYEN_IDR).toLocaleString("id-ID") + " IDR";
              break;
            case "IDRtoUSD":
              hasil = nominal + " IDR = " + (nominal / rateUSD_IDR).toFixed(2) + " USD";
              break;
            default:
              hasil = "Pilihan konversi tidak dikenali.";
          }
        }
        output.textContent = hasil;
      }
    </script>
